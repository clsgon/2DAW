Parte JS

var c = document.getElementById("canva");

var ctx = c.getContext("2d");

//Cuadricula de fondo

for(let x = 0.5; x < 500; x += 10)

{

ctx.moveTo(x, 0);

ctx.lineTo(x, 500);

ctx.stroke();

}

for(let y = 0.3; y < 500; y += 10)

{

ctx.moveTo(0, y);

ctx.lineTo(500, y);

ctx.stroke();

}

//Cuadrado verde esquina izquierda superior

ctx.fillStyle = "#05f335";

ctx.fillRect(0, 0, 175, 175); //fillRect(x, y, width, height);

//Configurar Sombra

ctx.shadowOffsetX = 10;

ctx.shadowOffsetY = 10;

ctx.shadowBlur = 5;

ctx.shadowColor = "rgba(0, 0, 0, 0.5)"

//Cuadrado con color degradado

var degradado = ctx.createLinearGradient(0, 0, 200, 380);

degradado.addColorStop(0, "#05f335");

degradado.addColorStop(1, "red");

ctx.fillStyle = degradado;

ctx.fillRect(200, 75, 75, 150);

//Circulo

ctx.beginPath();

ctx.fillStyle = "#0077ff";

ctx.arc(100, 300, 70, 0, 2 * Math.PI); //arc(x, y, size, 0, 2 * Math.PI);

ctx.fill();

//Dibujar linea cuadrática

ctx.beginPath();

ctx.moveTo(30, 350);

ctx.quadraticCurveTo(300, 350, 250, 200);

ctx.strokeStyle = "black";

ctx.lineWidth = 3;

ctx.stroke();

//Agregar texto

ctx.font = "Italic 35px Arial";

ctx.fillStyle = "black";

ctx.fillText("¡Hola mundo!", 50, 50);

ctx.font = "Italic 35px Arial";

ctx.fillStyle = "black";

ctx.fillText("XXX", 100, 100);

ctx.fillStyle = "#0077ff";

//Cuadrados en curva

for(j = 0; j < 6; j += 1)

{

ctx.fillRect(400, 80, 75, 75);

ctx.translate(50, 50);

ctx.rotate(Math.PI / 16);

ctx.scale(0.7, 0.7);

}
